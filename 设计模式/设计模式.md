## 1. 多态

同一个操作作用在不同的对象上产生不同的结果。

```javascript
function makeSound(animal) {
    animal.sound();
}

function Duck() {}
Duck.prototype.sound = function() {
    console.log("嘎嘎嘎");
}

function Chicken() {}
Chicken.prototype.sound = function() {
    console.log("咯咯咯")
}

makeSound(new Duck())	//"嘎嘎嘎"
makeSound(new Chicken())	//"咯咯咯"
```

## 2. 单例模式

保证一个类仅有一个实例，且该实例能访问它的全局访问点

```javascript
let Singletom = function(name) {
    this.name = name;
    this.instance = null;
}
Singletom.prototype.getName = function() {
    console.log(this.name);
}
Singletom.getInstance = function(name) {
    if (!this.instance){
        this.instance = new Singleton(name);
    }
    return this.instance
}
```

## 6. 策略模式

定义一系列的算法，把它们一个个封装起来，并且使它们可以相互替换。

策略模式至少有两部分组成：

1. 策略类，封装了具体的算法以及实现的具体过程
2. 环境类，根据不同的环境从策略类中选择不同的方法

```javascript
let strategies = {
    "S": function(salary) {
        return salary * 4;
    },
    "A": function(salary) {
        return salary * 3;
    }
}
let calculateBonus = function(level, salary) {
    return strategies[level](salary)
}
```

## 9. 发布订阅模式

又称观察者模式，定义对象间的一对多的依赖关系，当一个对象的状态发生改变时，所有依赖于它的对象都将得到通知。

```javascript
let event = {
    clientList: {},
    listen(key, fn) {
        if (!this.clientList[key]) {
            this.clientList[key] = [];
        }
        this.clientList[key].push(fn);
    },
    trigger() {
        let key = [].shift.call(arguments),
            fns = this.clientList[key];
        if (!fns || fns.length === 0){
            return false;
        }
        for (let i = 0, fn; fn = fns[i++];){
            fn.apply(this, arguments);
        }
    }
};

let installEvent = function(obj) {
    for (let i in event) {
        obj[i] = event[i];
    }
}

let salesOffices = {};
installEvent(salesOffices);
salesOffices.listen("squareMeter88", function(price) {
    console.log("价格=" + price)
})
salesOffices.trigget("squareMeter88", 200000);
```

## 10. 命令模式

```javascript
let Ryu = {
    attack() {
        console.log("攻击");
    },
    defense() {
        console.log("防御");
    },
    jump() {
        console.log("跳跃");
    },
    crouch() {
        console.log("蹲下");
    }
};
let makeCommand = function(receiver, state) {
    if (receiver[state]){
        return function() {
            receiver[state]();
        }
    }else {
        return null
    }
}

let commands = {
    "119": "jump",
    "115": "crouch",
    "97": "defense",
    "100": "attack"
}

let commandStack = [];
document.onkeypress = function (e) {
    let keyCode = e.keyCode,
        command = makeCommand(Ryu, commands[keyCode]);
    if(command) {
        command();
        commandStack.push(command);
    }
}
```
